prefix owl: <http://www.w3.org/2002/07/owl#> 
prefix rdf: <http://www.w3.org/1999/02/22-rdf-syntax-ns#> 
prefix rdfs: <http://www.w3.org/2000/01/rdf-schema#> 
prefix sio: <<http://semanticscience.org/resource/> 
prefix osm: <http://www.w3.org/2008/05/osm#> 

SELECT DISTINCT ?id ?admdate ?icd ?cadmission ?ag WHERE {
  ?id_E rdf:type osm:idPatient  ;
    sio:isAttributeOf    ?patient_V  ;
    sio:hasValue ?id .
	
  ?admdate_E rdf:type osm:admissionDate  ;
    sio:isAttributeOf    ?hmr_V  ;
    sio:existsAt     ?admission_V  ;
    sio:hasValue ?admdate .

  ?icd_E rdf:type osm:ICD10  ;
    sio:isAttributeOf    ?hmr_V  ;
    sio:hasValue ?icd .

  ?cadmission_E rdf:type osm:typeOfAdmission  ;
    sio:isAttributeOf    ?hmr_V  ;
    sio:hasValue ?cadmission .

  ?ag_E rdf:type osm:ageGroup  ;
    sio:isAttributeOf    ?hmr_V  ;
    sio:hasValue ?ag .

  ?patient_V rdf:type sio:Human  ;
    osm:hasHMR ?hmr_V .

  ?hmr_V rdf:type osm:HMR  ;
    osm:isHMROf ?patient_V .
}
